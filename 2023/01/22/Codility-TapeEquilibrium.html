import Foundation
import Glibc

public func solution(_ A : inout [Int]) -> Int {

    var max1 = A[0]
    var max2 = sum(A) - max1
    var min = abs(max1 - max2)
    for i in 1..<A.count-1{
        max1 = max1 + A[i]
        max2 = max2 - A[i]
        var target = abs(max1-max2)
        if min >= target{
            min = target
        }
    }
    return min
}
